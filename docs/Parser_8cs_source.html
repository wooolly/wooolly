156  object propertyValue = property.GetValue(target);
157  Type actualType = propertyValue?.GetType();
158  // Execute XPath to find all desirable elements
159  XmlNodeList desiredElems = currentNode.SelectNodes(attribute.Path, nsManager);
160  foreach (XmlNode desiredElem in desiredElems)
161  {
162  // Look for inner elements
163  if ((propertyType.IsClass || propertyType.IsInterface) &&
164  typeof(IParsable).IsAssignableFrom(propertyType))
165  {
