59 [ThrowIf("Locked", typeof(LockedLoadingException))]
60 public IManager Load(string dllPath)
61 {
62 // Load DLL (this will also load referenced assemblies)
63 System.Reflection.Assembly assembly = System.Reflection.Assembly.LoadFrom(dllPath);
64 DLLs.Add(assembly);
65
66 // Get all assembly names (including those referenced)
67 loadedAssemblies.AddRange(
68 assembly
69 .GetReferencedAssemblies()
70 .Select(n => n.FullName)
71 .Concat(new string[] { assembly.FullName }) // Add self (just loaded)
72 );
73
74 return this;
75 }

77 [ThrowIf("Locked", typeof(LockedLoadingException))]
78 public IManager LockLoading()
79 {
80 // Lock further loading
81 Locked = true;
82
83 // Cache exported types - all referenced assemblies
84 exportedTypes = AppDomain.CurrentDomain
85 .GetAssemblies()
86 // Exclude dynamic assemblies and those not loaded or referenced
87 .Where(a => !a.IsDynamic && loadedAssemblies.Contains(a.FullName))
88 // Exclude abstract/interface types
89 .SelectMany(a => a.GetExportedTypes().Where(t => !t.IsAbstract))
90 .ToArray();
91
92 return this;

138 public void CacheInterfaceType<T>(Type callableType, Type callerType) where T : ICache
139 {
140 if (!callableType.IsInterface)
141 throw new UnexpectedTypeException($"UnpackInterfaceType requires an interface type to be passed in, but {callableType.Name} was passed instead");
142 if (!callerType.IsInterface)
143 throw new UnexpectedTypeException($"UnpackInterfaceType requires an interface type to be passed in, but {callerType.Name} was passed instead");
144
145 // Get the exported types which match the target interface type
146 InterfaceTypes[callableType] = exportedTypes
147 .Where(t => callableType.IsAssignableFrom(t));
148
149 foreach (Type implementationType in InterfaceTypes[callableType])
150 {
151 // Cache implementation type
152 T cache = (T)Activator.CreateInstance(typeof(T));
153 ImplementationCache[implementationType] = cache.Unpack(implementationType, callerType);
154 // Raise event for implementation type
155 CacheInterfaceTypeCompleted?.Invoke(implementationType, cache);
156 }
157 }
