128 private void UnpackMethods(Type callableType, Type callerType)
129 {
130 // Dissect (exported methods and target methods)
131 MethodInfo[] exportedMethods = callableType.GetMethods(BindingFlags.Instance | BindingFlags.NonPublic);
132 PropertyInfo[] targetMethods = callerType.GetProperties(BindingFlags.Instance | BindingFlags.Public)
133 .Where(p => typeof(Delegate).IsAssignableFrom(p.PropertyType))
134 .ToArray();
135 // Cache
136 foreach (PropertyInfo targetMethod in targetMethods)
137 {
138 // Raises error if found more than once! Continues loop if not found
139 // (doesn't currently support overloading)
140 MethodInfo[] matchingExportedMethods = exportedMethods.Where(m => m.Name == targetMethod.Name).ToArray();
141 if (matchingExportedMethods.Length > 1)
142 throw new Exceptions.UnsupportedOverloadException($"{targetMethod.Name} in {callableType.Name} is overloaded, which is unsupported in Maker: Online");
143 else if (matchingExportedMethods.Length < 1)
144 continue;
145 MethodInfo matchingExportedMethod = matchingExportedMethods[0];
146
147 // Build up expression tree to compile custom delegate...
148 // Add "this" param (because open instance delegate)
149 ParameterExpression instanceParam = Expression.Parameter(targetMethod.PropertyType.GetGenericArguments().First(), "this");
150 // Add all other required arguments defined by targetMethod method and exported method
151 ParameterExpression[] genericExpressionArgs = GetGenericExpressionArgs(targetMethod);
152 ConstantExpression[] dependencyExpressionArgs = GetDependencyExpressionArgs(genericExpressionArgs.Length, matchingExportedMethod);
153 // Define method call, using all expressions
154 MethodCallExpression methodCallExpression = Expression.Call(
155 Expression.Convert(instanceParam, matchingExportedMethod.DeclaringType),
156 matchingExportedMethod,
157 new Expression[] { }.Concat(genericExpressionArgs).Concat(dependencyExpressionArgs));
158 // Compile to delegate
159 Methods[targetMethod.Name] = Expression.Lambda(
160 targetMethod.PropertyType,
161 methodCallExpression,
162 new ParameterExpression[] { instanceParam }.Concat(genericExpressionArgs))
163 .Compile();
164 }
165 }

175 private void UnpackConstructor(Type callableType, Type _)
176 {
177 // Dissect
178 ConstructorInfo exportedConstructor;
179 try
180 {
181 exportedConstructor = callableType
182 .GetConstructors(BindingFlags.Instance | BindingFlags.NonPublic)
183 .Single();
184 }
185 catch (InvalidOperationException)
186 {
187 throw new Exceptions.InvalidImplementationException($"{callableType.Name} should have a single non-public constructor");
188 }
189
190 PropertyInfo targetConstructor = callerType
191 .GetPropertyExt(CONSTRUCTOR_PROPERTY, BindingFlags.Instance | BindingFlags.Public);
192
193 // Build expression tree...
194 // Define constant expressions for all dependency arguments
195 ParameterExpression[] genericExpressionArgs = GetGenericExpressionArgs(targetConstructor);
196 ConstantExpression[] dependencyExpressionArgs = GetDependencyExpressionArgs(0, exportedConstructor);
197 // Define constructor call (with dependency arguments)
198 NewExpression newExpression = Expression.New(exportedConstructor, dependencyExpressionArgs);
199 // Define initialization for instance members
200 MemberBinding[] memberBindings = GetInitializerExpressions(targetConstructor, genericExpressionArgs);
201 MemberInitExpression initExpression = Expression.MemberInit(newExpression, memberBindings);
202
203 // Compile and cache
204 Delegate constructorDelegate = Expression.Lambda(targetConstructor.PropertyType, initExpression, genericExpressionArgs).Compile();
205 Constructor = constructorDelegate;
206 }
